<script>

const INVEST = 250;

const MONTH_NAMES = ['Jan','Fev','Mar','Abr','Mai','Jun','Jul','Ago','Set','Out','Nov','Dez'];

const FARM_NORMAL = 35;


let spStart=3, spLucro=99, spChart=null;

let farmStart=3, farmJun=70, farmJul=95, farmPos=35, farmChart=null;


function calcRows(startMonth, getLucro) {

  const rows=[]; let acum=0;

  const obraEnd=startMonth+3; let done=false;

  for(let i=startMonth; i<30&&!done; i++){

    const mi=i%12, fase=i<obraEnd?'obra':'pos';

    const lucro=getLucro(mi,fase);

    const prev=acum; acum+=lucro;

    rows.push({month:MONTH_NAMES[mi],mi,fase,lucro,acum,falta:Math.max(0,INVEST-acum),breakeven:prev<INVEST&&acum>=INVEST});

    if(acum>=INVEST) done=true;

    if(rows.length>16) done=true;

  }

  return rows;

}


function makeChart(canvasId, rows, getColor) {

  return new Chart(document.getElementById(canvasId),{

    type:'bar',

    data:{labels:rows.map(r=>r.month),datasets:[

      {label:'Acumulado',data:rows.map(r=>Math.round(r.acum)),backgroundColor:rows.map(getColor),borderRadius:4,order:2},

      {label:'Meta',data:Array(rows.length).fill(INVEST),type:'line',borderColor:'#E24B4A',borderWidth:2,borderDash:[6,4],pointRadius:0,fill:false,order:1}

    ]},

    options:{

      responsive:true,maintainAspectRatio:false,animation:{duration:250},

      plugins:{legend:{display:false},tooltip:{callbacks:{

        label:ctx=>ctx.datasetIndex===0?(ctx.parsed.y>=INVEST?`Acumulado: R$${ctx.parsed.y}k ✓`:`Acumulado: R$${ctx.parsed.y}k`):`Meta: R$${ctx.parsed.y}k`,

        afterLabel:ctx=>ctx.datasetIndex===0&&ctx.parsed.y<INVEST?`Falta: R$${INVEST-ctx.parsed.y}k`:''

      }}},

      scales:{

        x:{grid:{display:false},ticks:{font:{size:12}}},

        y:{grid:{color:'rgba(0,0,0,0.06)'},max:400,ticks:{font:{size:12},callback:v=>`R$${v}k`}}

      }

    }

  });

}


function refreshChart(chart,rows,getColor){

  chart.data.labels=rows.map(r=>r.month);

  chart.data.datasets[0].data=rows.map(r=>Math.round(r.acum));

  chart.data.datasets[0].backgroundColor=rows.map(getColor);

  chart.data.datasets[1].data=Array(rows.length).fill(INVEST);

  chart.update();

}


function updateSP(){

  spLucro=parseInt(document.getElementById('sp-lucro').value);

  document.getElementById('sp-lucro-val').textContent=`R$${spLucro}k`;

  const rows=calcRows(spStart,()=>spLucro);

  const durante=spLucro*3;

  const restante=Math.max(0,INVEST-durante);

  const total=rows.length;

  document.getElementById('sp-v-durante').textContent=`R$${durante}k`;

  document.getElementById('sp-v-total').textContent=`${total} ${total===1?'mês':'meses'}`;

  document.getElementById('sp-v-restante').textContent=restante===0?'R$0k — quitado':`R$${restante}k`;

  document.getElementById('sp-s-restante').textContent=restante===0?'100% quitado durante obra':`250k − ${durante}k`;

  document.getElementById('sp-insight').innerHTML=

    `SP iniciando em <strong>${MONTH_NAMES[spStart%12]}</strong> com <strong>R$${spLucro}k/mês</strong>. `+

    `Recupera <strong>R$${durante}k durante a obra</strong>. `+

    (restante===0

      ?`<strong style="color:#1D9E75;">Investimento 100% quitado durante a obra!</strong>`

      :`Restante de <strong>R$${restante}k</strong> quitado em <strong>${total-3} ${total-3===1?'mês':'meses'}</strong> pós-obra.`)+

    ` Payback total: <strong>${total} ${total===1?'mês':'meses'}</strong>.`;

  const getColor=r=>r.fase==='obra'?'#85B7EB':r.acum>=INVEST?'#1D9E75':'#B5D4F4';

  if(!spChart){spChart=makeChart('sp-chart',rows,getColor);}else{refreshChart(spChart,rows,getColor);}

  const tbody=document.getElementById('sp-tbody');

  tbody.innerHTML='';

  rows.forEach(r=>{

    const tr=document.createElement('tr');

    if(r.breakeven)tr.style.background='#f0faf4';

    const badge=r.fase==='obra'?'<span class="badge obra-sp">Obra</span>':r.breakeven?'<span class="badge breakeven">Break-even ✓</span>':'<span class="badge pos-sp">Pós-obra</span>';

    tr.innerHTML=`<td style="font-weight:500;">${r.month}</td><td>${badge}</td><td>R$${Math.round(r.lucro)}k</td><td>R$${Math.round(r.acum)}k</td><td style="color:${r.falta===0?'#1D9E75':'#aaa'};">${r.falta===0?'—':'R$'+Math.round(r.falta)+'k'}</td>`;

    tbody.appendChild(tr);

  });

}


function setStartSP(m,btn){

  spStart=m;

  document.querySelectorAll('#sp-start-tabs .tab').forEach(t=>t.classList.remove('active','sp'));

  btn.classList.add('active','sp');

  updateSP();

}


function updateFarm(){

  farmJun=parseInt(document.getElementById('farm-jun').value);

  farmJul=parseInt(document.getElementById('farm-jul').value);

  farmPos=parseInt(document.getElementById('farm-pos').value);

  document.getElementById('farm-jun-val').textContent=`R$${farmJun}k`;

  document.getElementById('farm-jul-val').textContent=`R$${farmJul}k`;

  document.getElementById('farm-pos-val').textContent=`R$${farmPos}k`;

  const rows=calcRows(farmStart,(mi,fase)=>{

    if(mi===5)return farmJun;

    if(mi===6)return farmJul;

    return fase==='obra'?FARM_NORMAL:farmPos;

  });

  const durante=rows.filter(r=>r.fase==='obra').reduce((s,r)=>s+r.lucro,0);

  const junjul=farmJun+farmJul;

  const diff=junjul-95;

  document.getElementById('farm-v-durante').textContent=`R$${Math.round(durante)}k`;

  document.getElementById('farm-v-total').textContent=`${rows.length} ${rows.length===1?'mês':'meses'}`;

  document.getElementById('farm-v-junjul').textContent=`R$${junjul}k`;

  document.getElementById('farm-s-junjul').textContent=diff>=0?`+R$${diff}k vs Jul 2025 (R$95k)`:`-R$${Math.abs(diff)}k vs Jul 2025 (R$95k)`;

  document.getElementById('farm-insight').innerHTML=

    `Farm iniciando em <strong>${MONTH_NAMES[farmStart%12]}</strong>. `+

    `Meses normais de obra: <strong>R$${FARM_NORMAL}k</strong>. `+

    `Jun + Jul 2026 somam <strong>R$${junjul}k</strong> `+

    `(${diff>=0?`<strong style="color:#1D9E75;">+R$${diff}k acima</strong>`:`<strong style="color:#E24B4A;">R$${Math.abs(diff)}k abaixo</strong>`} do R$95k de Jul 2025). `+

    `Break-even em <strong>${rows.length} meses</strong> do início da obra.`;

  const getColor=r=>{

    if(r.mi===5)return'#EF9F27';

    if(r.mi===6)return'#378ADD';

    if(r.fase==='obra')return'#9FE1CB';

    return r.acum>=INVEST?'#1D9E75':'#B5D4F4';

  };

  if(!farmChart){farmChart=makeChart('farm-chart',rows,getColor);}else{refreshChart(farmChart,rows,getColor);}

  const tbody=document.getElementById('farm-tbody');

  tbody.innerHTML='';

  rows.forEach(r=>{

    const tr=document.createElement('tr');

    if(r.breakeven)tr.style.background='#f0faf4';

    let badge='',cor='#1a1a1a',fw='400';

    if(r.mi===5){badge=`<span class="badge junho-ev">${r.fase==='obra'?'Obra + ':''}Retiro Vox + FWS</span>`;cor='#BA7517';fw='500';}

    else if(r.mi===6){badge=`<span class="badge julho-ev">${r.fase==='obra'?'Obra + ':''}2× Greenhouse + ZBreak</span>`;cor='#185FA5';fw='500';}

    else if(r.fase==='obra'){badge='<span class="badge obra-f">Obra</span>';}

    else if(r.breakeven){badge='<span class="badge breakeven">Break-even ✓</span>';}

    else{badge='<span class="badge pos-f">Pós-obra</span>';}

    tr.innerHTML=`<td style="font-weight:500;">${r.month}</td><td>${badge}</td><td style="color:${cor};font-weight:${fw};">R$${Math.round(r.lucro)}k</td><td>R$${Math.round(r.acum)}k</td><td style="color:${r.falta===0?'#1D9E75':'#aaa'};">${r.falta===0?'—':'R$'+Math.round(r.falta)+'k'}</td>`;

    tbody.appendChild(tr);

  });

}


function setStartFarm(m,btn){

  farmStart=m;

  document.querySelectorAll('#farm-start-tabs .tab').forEach(t=>t.classList.remove('active','farm'));

  btn.classList.add('active','farm');

  updateFarm();

}


function switchView(v,btn){

  document.querySelectorAll('.view').forEach(el=>el.classList.remove('active'));

  document.querySelectorAll('.view-btn').forEach(el=>el.classList.remove('active','sp','farm'));

  document.getElementById('view-'+v).classList.add('active');

  btn.classList.add('active',v);

}
